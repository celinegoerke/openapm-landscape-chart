1. Um auf die Petclinic zugreifen zu können, bitte folgende Anweisungen verwenden:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "helmchartbachelor.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Besuchen Sie http://127.0.0.1:8080 um Ihre Anwendung zu verwenden"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:8080

2. Um auf den Configurationsserver von inspectIT zugreifen zu können, bitte folgende Anweisungen verwenden:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name=configserver,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Besuchen Sie http://127.0.0.1:8090 um den Configurationsserver zu verwenden"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8090:8090

3. Um auf den Prometheus Server zugreifen zu können, bitte folgende Anweisungen verwenden:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app=prometheus,component=server" -o jsonpath="{.items[0].metadata.name}")
  echo "Besuchen Sie http://127.0.0.1:9090 um den Prometheus Server zu verwenden"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9090:9090

4. Um auf Grafana zugreifen zu können, bitte folgende Anweisungen verwenden:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name=grafana,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Besuchen Sie http://127.0.0.1:3000 um Grafana zu verwenden"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:3000

  Das Passwort von Grafana wird mit folgender Anweisung ausgegeben:
  kubectl get secret --namespace {{ .Release.Namespace }} {{ .Release.Name }}-grafana -o jsonpath="{.data.admin-password}" | base64 --decode ; echo

  Zugriff zu Grafana wird mit dem ausgelesenen Passwort und dem Usernamen "admin" gewährt.
